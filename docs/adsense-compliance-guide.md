// utils/contentModeration.ts
const PROHIBITED_KEYWORDS = [
  'adult', 'explicit', 'nude', 'sexual', 'porn',
  'violence', 'gore', 'blood', 'kill', 'murder',
  'hate', 'racist', 'discrimination',
  'drugs', 'cocaine', 'heroin', 'meth',
  'weapons', 'guns', 'bombs', 'explosives',
  'hack', 'malware', 'virus',
  'counterfeit', 'fake',
  // Add more as needed
];

export function moderateContent(prompt: string): {
  isAllowed: boolean;
  reason?: string;
} {
  const lowerPrompt = prompt.toLowerCase();
  
  for (const keyword of PROHIBITED_KEYWORDS) {
    if (lowerPrompt.includes(keyword)) {
      return {
        isAllowed: false,
        reason: `Content contains prohibited keywords. Please review our content policy.`
      };
    }
  }
  
  return { isAllowed: true };
}
